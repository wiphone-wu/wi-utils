# wi-utils
Common js functions organized by wiphone.

# install
npm install wi-utils

# use
## use in vue
* import in entry file
* import wi from 'wi-utils';
* Vue.prototype.$wi = wi;
### use in page
* this.$wi.funs();

## use in js
### import and use in the page
* import wi from 'wi-utils';
* wi.funs();
